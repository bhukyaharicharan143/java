import java.util.*; public class StatsCalculator { public static void main(String[] args) { int[] data = {10, 20, 20, 30, 40, 50, 50, 50, 60}; // Mean double sum = 0; for (int num : data) { sum += num; } double mean = sum / data.length; // Median Arrays.sort(data); double median; if (data.length % 2 == 0) { median = (data[data.length / 2 - 1] + data[data.length / 2]) / 2.0; } else { median = data[data.length / 2]; } // Mode Map freqMap = new HashMap<>(); for (int num : data) { freqMap.put(num, freqMap.getOrDefault(num, 0) + 1); } int mode = data[0]; int maxCount = 0; for (Map.Entry entry : freqMap.entrySet()) { if (entry.getValue() > maxCount) { mode = entry.getKey(); maxCount = entry.getValue(); } } // Output System.out.println("Data: " + Arrays.toString(data)); System.out.println("Mean: " + mean); System.out.println("Median: " + median); System.out.println("Mode: " + mode + " (Count: " + maxCount + ")");
